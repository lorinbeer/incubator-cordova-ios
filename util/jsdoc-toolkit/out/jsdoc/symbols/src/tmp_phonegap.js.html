204      * getting the position data.
205      * @param {PositionOptions} options The options for getting the position data
206      * such as timeout and the frequency of the watch.
207      */
208     Geolocation.prototype.watchPosition = function(successCallback, errorCallback, options) {
209     	// Invoke the appropriate callback with a new Position object every time the implementation 
210     	// determines that the position of the hosting device has changed. 
211     	
212     	this.getCurrentPosition(successCallback, errorCallback, options);
213     	var frequency = (options != undefined)? options.frequency : 10000;
214     	
215     	var that = this;
216     	return setInterval(function() {
217     		that.getCurrentPosition(successCallback, errorCallback, options);
218     		//navigator.geolocation.getCurrentPosition(successCallback, errorCallback, options);
219     	}, frequency);
220     }

340     /**
341      * This class contains position information.
342      * @param {Object} lat
343      * @param {Object} lng
344      * @param {Object} acc
345      * @param {Object} alt
346      * @param {Object} altacc
347      * @param {Object} head
348      * @param {Object} vel
349      * @constructor
350      */
351     function Position(lat, lng, acc, alt, altacc, head, vel) {
352     	/**
353     	 * The latitude of the position.
354     	 */
355     	this.latitude = lat;
356     	/**
357     	 * The longitude of the position,
358     	 */
359     	this.longitude = lng;
360     	/**
361     	 * The accuracy of the position.
362     	 */
363     	this.accuracy = acc;
364     	/**
365     	 * The altitude of the position.
366     	 */
367     	this.altitude = alt;
368     	/**
369     	 * The altitude accuracy of the position.
370     	 */
371     	this.altitudeAccuracy = altacc;
372     	/**
373     	 * The direction the device is moving at the position.
374     	 */
375     	this.heading = head;
376     	/**
377     	 * The velocity with which the device is moving at the position.
378     	 */
379     	this.velocity = vel;
380     	/**
381     	 * The time that the position was obtained.
382     	 */
383     	this.timestamp = new Date().getTime();
384     }
