  3      * This represents the PhoneGap API itself, and provides a global namespace for accessing
  4      * information about the state of PhoneGap.
  5      * @class
  6      */
  7     PhoneGap = {
  8         queue: {
  9             ready: true,
 10             commands: [],
 11             timer: null
 12         },
 13         _constructors: []
 14     };
 15     
 16     /**
 17      * Boolean flag indicating if the PhoneGap API is available and initialized.
 18      */
 19     PhoneGap.available = DeviceInfo.uuid != undefined;
 20     
 21     /**
 22      * Add an initialization function to a queue that ensures it will run and initialize
 23      * application constructors only once PhoneGap has been initialized.
 24      * @param {Function} func The function callback you want run once PhoneGap is initialized
 25      */
 26     PhoneGap.addConstructor = function(func) {
 27         var state = document.readyState;
 28         if (state != 'loaded' && state != 'complete')
 29             PhoneGap._constructors.push(func);
 30         else
 31             func();
 32     };
 33     (function() {
 34         var timer = setInterval(function() {
 35             var state = document.readyState;
 36             if (state != 'loaded' && state != 'complete')
 37                 return;
 38             clearInterval(timer);
 39             while (PhoneGap._constructors.length > 0) {
 40                 var constructor = PhoneGap._constructors.shift();
 41                 try {
 42                     constructor();
 43                 } catch(e) {
 44                     if (typeof(debug['log']) == 'function')
 45                         debug.log("Failed to run constructor: " + e.message);
 46                     else
 47                         alert("Failed to run constructor: " + e.message);
 48                 }
 49             }
 50         }, 1);
 51     })();

 54     /**
 55      * Execute a PhoneGap command in a queued fashion, to ensure commands do not
 56      * execute with any race conditions, and only run when PhoneGap is ready to
 57      * recieve them.
 58      * @param {String} command Command to be run in PhoneGap, e.g. "ClassName.method"
 59      * @param {String[]} [args] Zero or more arguments to pass to the method
 60      */
 61     PhoneGap.exec = function() {
 62         PhoneGap.queue.commands.push(arguments);
 63         if (PhoneGap.queue.timer == null)
 64             PhoneGap.queue.timer = setInterval(PhoneGap.run_command, 10);
 65     };
 66     /**
 67      * Internal function used to dispatch the request to PhoneGap.  It processes the command
 68      * queue and executes the next command on the list.
 69      * @private
 70      */
 71     PhoneGap.run_command = function() {
 72         if (!PhoneGap.available)
 73             return;
 74     
 75         PhoneGap.queue.ready = false;
 76     
 77         var args = PhoneGap.queue.commands.shift();
 78         if (PhoneGap.queue.commands.length == 0) {
 79             clearInterval(PhoneGap.queue.timer);
 80             PhoneGap.queue.timer = null;
 81         }
 82     
 83         var uri = [];
 84         for (var i = 1; i < args.length; i++) {
 85             uri.push(encodeURIComponent(args[i]));
 86         }
 87         document.location = "gap://" + args[0] + "/" + uri.join("/");
 88     
 89     };

 91     /**
 92      * this represents the mobile device, and provides properties for inspecting the model, version, UUID of the
 93      * phone, etc.
 94      * @constructor
 95      */
 96     function Device() {
 97         this.available = PhoneGap.available;
 98         this.model     = null;
 99         this.version   = null;
100         this.gap       = null;
101         this.uuid      = null;
102         try {
103             if (window['DroidGap'] != undefined && window.DroidGap.exists()) {
104                 this.available = true;
105                 this.isAndroid = true;
106                 this.uuid = window.DroidGap.getUuid();
107                 this.gapVersion = window.DroidGap.getVersion();
108             } else {          
109                 this.model     = DeviceInfo.platform;
110                 this.version   = DeviceInfo.version;
111                 this.gap       = DeviceInfo.gap;
112                 this.uuid      = DeviceInfo.uuid;
113             }
114         } catch(e) {
115             this.available = false;
116         }
117     }

334     /**
335      * Utility function for rendering and indenting strings, or serializing
336      * objects to a string capable of being printed to the console.
337      * @param {Object|String} message The string or object to convert to an indented string
338      * @private
339      */
340     Console.prototype.processMessage = function(message) {
341         if (typeof(message) != 'object') {
342             return encodeURIComponent(message);
343         } else {
344             /**
345              * @function
346              * @ignore
347              */
348             function indent(str) {
349                 return str.replace(/^/mg, "    ");
350             }
351             /**
352              * @function
353              * @ignore
354              */
355             function makeStructured(obj) {
356                 var str = "";
357                 for (var i in obj) {
358                     try {
359                         if (typeof(obj[i]) == 'object') {
360                             str += i + ":\n" + indent(makeStructured(obj[i])) + "\n";
361                         } else {
362                             str += i + " = " + indent(String(obj[i])).replace(/^    /, "") + "\n";
363                         }
364                     } catch(e) {
365                         str += i + " = EXCEPTION: " + e.message + "\n";
366                     }
367                 }
368                 return str;
369             }
370             return encodeURIComponent("Object:\n" + makeStructured(message));
371         }
372     };
